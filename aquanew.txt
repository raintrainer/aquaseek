<body>
    <div id="aquarium-container">
        <div class="water-surface"></div>
        <div class="aquarium-bottom"></div>
        <!-- Fische und Pflanzen werden per JavaScript hinzugefügt -->
        
        <div class="controls">
            <h2>Virtuelles Aquarium</h2>
            <button id="add-fish">Fisch hinzufügen</button>
            <button id="add-bubbles">Mehr Blasen</button>
            <p>Fische: <span id="fish-count">0</span></p>
        </div>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const aquarium = document.getElementById('aquarium-container');
            const fishCountElement = document.getElementById('fish-count');
            let fishCount = 0;
            
            // Fischtypen mit verschiedenen Farben und Größen
            const fishTypes = [
                { color: '#FF5733', size: { width: 80, height: 40 }, tailShape: 'triangle' },
                { color: '#33FF57', size: { width: 100, height: 50 }, tailShape: 'rounded' },
                { color: '#3357FF', size: { width: 60, height: 30 }, tailShape: 'triangle' },
                { color: '#F3FF33', size: { width: 120, height: 60 }, tailShape: 'rounded' },
                { color: '#FF33F3', size: { width: 70, height: 35 }, tailShape: 'triangle' }
            ];
            
            // Pflanzen hinzufügen
            function addPlants() {
                for (let i = 0; i < 5; i++) {
                    const plant = document.createElement('div');
                    plant.className = 'plant';
                    
                    // Zufällige Position am Boden
                    const left = Math.random() * 90 + 5;
                    const height = Math.random() * 100 + 50;
                    
                    plant.style.left = `${left}%`;
                    plant.style.height = `${height}px`;
                    plant.style.width = `${Math.random() * 40 + 20}px`;
                    
                    // Zufällige Farbe für die Pflanze
                    const hue = Math.random() * 60 + 100; // Grüntöne
                    plant.style.backgroundColor = `hsl(${hue}, 70%, 30%)`;
                    plant.style.borderRadius = '50% 50% 0 0';
                    
                    aquarium.appendChild(plant);
                }
            }
            
            // Fisch erstellen
            function createFish() {
                const fish = document.createElement('div');
                fish.className = 'fish';
                
                // Zufälligen Fischtyp auswählen
                const fishType = fishTypes[Math.floor(Math.random() * fishTypes.length)];
                
                fish.style.width = `${fishType.size.width}px`;
                fish.style.height = `${fishType.size.height}px`;
                
                // Fisch-Position
                const left = Math.random() * 80 + 10;
                const top = Math.random() * 70 + 10;
                
                fish.style.left = `${left}%`;
                fish.style.top = `${top}%`;
                
                // Fischrichtung (nach links oder rechts)
                const direction = Math.random() > 0.5 ? 1 : -1;
                fish.style.transform = `scaleX(${direction})`;
                
                // Fisch mit SVG zeichnen
                fish.innerHTML = `
                    <svg viewBox="0 0 100 60" width="100%" height="100%">
                        <path d="${getFishPath(fishType)}" fill="${fishType.color}" />
                        <circle cx="${direction > 0 ? 75 : 25}" cy="30" r="3" fill="black" />
                    </svg>
                `;
                
                // Schwimmanimation
                setInterval(() => {
                    const x = (Math.random() * 20 - 10);
                    const y = (Math.random() * 20 - 10);
                    
                    // Neue Position berechnen
                    let newLeft = parseFloat(fish.style.left) + x * 0.1;
                    let newTop = parseFloat(fish.style.top) + y * 0.1;
                    
                    // Innerhalb des Aquariums bleiben
                    newLeft = Math.max(5, Math.min(85, newLeft));
                    newTop = Math.max(5, Math.min(75, newTop));
                    
                    // Richtung ändern wenn nötig
                    if (x > 0 && direction < 0) {
                        fish.style.transform = `scaleX(1)`;
                    } else if (x < 0 && direction > 0) {
                        fish.style.transform = `scaleX(-1)`;
                    }
                    
                    fish.style.left = `${newLeft}%`;
                    fish.style.top = `${newTop}%`;
                }, 1000);
                
                aquarium.appendChild(fish);
                fishCount++;
                fishCountElement.textContent = fishCount;
                
                return fish;
            }
            
            // Fisch-Pfad basierend auf Typ
            function getFishPath(fishType) {
                if (fishType.tailShape === 'triangle') {
                    return "M10,30 Q30,10 50,30 Q70,50 90,30 Q70,55 50,30 Q30,5 10,30";
                } else {
                    return "M10,30 Q30,15 50,30 Q70,45 90,30 Q75,50 50,30 Q25,50 10,30";
                }
            }
            
            // Blasen erstellen
            function createBubbles(count) {
                for (let i = 0; i < count; i++) {
                    setTimeout(() => {
                        const bubble = document.createElement('div');
                        bubble.className = 'bubble';
                        
                        const size = Math.random() * 15 + 5;
                        const left = Math.random() * 100;
                        
                        bubble.style.width = `${size}px`;
                        bubble.style.height = `${size}px`;
                        bubble.style.left = `${left}%`;
                        bubble.style.bottom = `${Math.random() * 20 + 5}%`;
                        
                        // Animationsdauer basierend auf Größe
                        const duration = Math.random() * 6 + 4;
                        bubble.style.animationDuration = `${duration}s`;
                        
                        aquarium.appendChild(bubble);
                        
                        // Blase nach der Animation entfernen
                        setTimeout(() => {
                            bubble.remove();
                        }, duration * 1000);
                    }, Math.random() * 2000);
                }
            }
            
            // Event Listener für Buttons
            document.getElementById('add-fish').addEventListener('click', createFish);
            document.getElementById('add-bubbles').addEventListener('click', () => createBubbles(10));
            
            // Initiale Elemente erstellen
            addPlants();
            createBubbles(20);
            
            // Einige Fische am Anfang
            for (let i = 0; i < 3; i++) {
                setTimeout(createFish, i * 1000);
            }
            
            // Regelmäßig Blasen erstellen
            setInterval(() => createBubbles(5), 3000);
        });
    </script>
